import { Component } from '@angular/core';
import { DialogRef } from '@ngneat/dialog';

@Component({
  selector: 'app-add-contact',
  templateUrl: './add-contact.component.html',
  styleUrls: ['./add-contact.component.scss'],
})
export class AddContactComponent {
  name = '';
  phone = '';

  constructor(public ref: DialogRef) {}

  add() {
    if (!this.phone || !this.name) return;
    this.ref.close({ name: this.name, phone: this.phone });
  }
}

<div class="p-4">
  <h3 class="text-2xl mb-6">Add New Contact</h3>
  <input
    [(ngModel)]="name"
    class="form-control rounded-md mb-3"
    type="text"
    placeholder="Name"
  />
  <input
    [(ngModel)]="phone"
    class="form-control rounded-md mb-6"
    type="number"
    placeholder="Phone"
  />

  <button class="btn btn-primary mt-4 rounded-lg" (click)="add()">
    <i class="ti-plus"></i> Add Contact
  </button>
</div>
